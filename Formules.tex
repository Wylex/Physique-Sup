\subsection{Onde progressive sinuso\"idale $^\thect$}
\begin{enumerate}
	\item $s(x,t) = A\cos(\omega t - kx + \varphi)$

	\item $s(x,t) = C\cos(kx + \psi)\cos(\omega t + \varphi)$
	\item $L = n \frac{\lambda}{2}$
\end{enumerate}
